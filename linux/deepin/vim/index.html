#设置语法高亮
echo 'syntax on' >> ~/.vimrc

#开启行号显示【标准方式】或上面的【简写方式】也可以
echo 'set number' >> ~/.vimrc
#开启行号显示【简写方式】或下面的【标准方式】也可以
echo 'set nu' >> ~/.vimrc

#开启拼写检查
echo 'set spell' >> ~/.vimrc

#重点：关闭鼠标模式
echo 'set mouse=""' >> ~/.vimrc
