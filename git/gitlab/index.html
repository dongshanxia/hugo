# 在工作目录下创建 docker-compose.yml 文件，编辑以下内容
version: "3"
services:
  gitlab:
    image: 'gitlab/gitlab-ce:latest'
    restart: always
    container_name: gitlab
    ports:
      - '80:80'
    volumes:
      - '/docker/software/gitlab/config:/etc/gitlab'
      - '/docker/software/gitlab/logs:/var/log/gitlab'
      - '/docker/software/gitlab/data:/var/opt/gitlab'
    environment:
      TZ: 'Asia/Shanghai'
      GITLAB_OMNIBUS_CONFIG: |
                external_url 'http://127.0.0.1:80'
#         Add any other gitlab.rb configuration here, each on its own line
         gitlab_rails['gitlab_ssh_host'] = '127.0.0.1'
#         gitlab_rails['gitlab_shell_ssh_port'] = 26013
